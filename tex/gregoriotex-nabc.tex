% Dimension for communicating NABC left overflow from Lua to TeX.
% When nabc alignment is 'neume', left-side significative letters extend
% to the left of the neume.  This dimension holds the overflow width so
% that \gre@typesettextabovelines can reserve space at the note level.
\newdimen\gre@dimen@nabcleftoverflow
\gre@dimen@nabcleftoverflow=0pt\relax
% Tracks how much kern has been emitted at note level across all NABC voices.
% Used to coordinate overflow between voice 1 (above) and voice 2 (below).
\newdimen\gre@dimen@nabckernemitted
\gre@dimen@nabckernemitted=0pt\relax
\newbox\gre@box@nabctemp
% Boxes for saving NABC voice content between prepare and place phases.
% The two-phase processing ensures that all voices are pre-evaluated and
% all note-level kerns emitted before any voice hbox is placed, so that
% each voice is at the correct horizontal position (P + total_kern).
\newbox\gre@box@nabc@voice@i
\newbox\gre@box@nabc@voice@ii

% \gresetnabcfont[#1]#2#3
% #1: voice
% #2: font name
% #3: font size
\def\gresetnabcfont{\@ifnextchar[{\gre@setnabcvoicefont}{\gre@setnabcvoicefont[1]}}%

\def\gre@setnabcvoicefont[#1]#2#3{%
  \expandafter\xdef\csname gre@nabcvoice@\romannumeral#1@fontname\endcsname{#2}%
  \expandafter\font\csname gre@font@nabcvoice@\romannumeral#1\endcsname={name:#2} at #3 pt\relax
  \global\font\gre@font@nabc={name:#2} at #3 pt\relax %
  {%
    \csname gre@font@nabcvoice@\romannumeral#1\endcsname
    \directlua{gregoriotex.init_nabc_font("\luaescapestring{#2}")}%
  }%
}

% #1: NABC string
% #2: font name
% #3: scale
% #4: voice number
\def\gre@nabccharno#1#2#3#4{%
  \gre@trace{gre@nabccharno{#1}{#2}{#3}{#4}}%
  {\directlua{gregoriotex.print_nabc(gregoriotex.parse_nabc("#1", "\luaescapestring{#2}", #3, #4))}}%
  \gre@trace@end%
}

% \GreNABCChar[#1]#2
% #1: voice
% #2: char
\def\GreNABCChar{\@ifnextchar[{\gre@nabcvoicechar}{\gre@nabcvoicechar[1]}}%
\def\gre@nabcvoicechar[#1]#2{{%
  %\catcode`\~=12{}%
  %\catcode`\@=11{}%
  \csname gre@font@nabcvoice@\romannumeral#1\endcsname
  \gre@style@nabc
  \ifcase#1\relax\or\gre@style@abovelinesnabc\or\gre@style@belowlinesnabc\fi
  \gre@nabccharno{#2}{\csname gre@nabcvoice@\romannumeral#1@fontname\endcsname}{1}{#1}%
  \ifcase#1\relax\or\endgre@style@abovelinesnabc\or\endgre@style@belowlinesnabc\fi
  \endgre@style@nabc
}}

% Per-voice 'skip-alterations' flags: when true, NABC neumes are anchored to
% the first real square note, skipping any leading alteration signs.
\newif\ifgre@nabc@voice@i@skipalterations
\gre@nabc@voice@i@skipalterationsfalse
\newif\ifgre@nabc@voice@ii@skipalterations
\gre@nabc@voice@ii@skipalterationsfalse
% In-alteration signal: set to true by \gre@alteration@visible around the
% \gre@newglyphcommon call so that deferred pre/emit macros can detect they
% are being called from an alteration and defer execution.
\newif\ifgre@nabc@in@alteration
\gre@nabc@in@alterationfalse
% Per-voice 'ready' flags: set when the deferred pre-phase fills the nabc
% box, cleared after the emit-phase places it.
\newif\ifgre@nabc@voice@i@ready
\gre@nabc@voice@i@readyfalse
\newif\ifgre@nabc@voice@ii@ready
\gre@nabc@voice@ii@readyfalse
% Per-voice 'altskipped' flags: set when the deferred pre-phase detects
% an alteration.  When the real note is reached, these flags cause the
% left-overflow kern to be suppressed (the accidental already provides
% visual space before the note).
\newif\ifgre@nabc@voice@i@altskipped
\gre@nabc@voice@i@altskippedfalse
\newif\ifgre@nabc@voice@ii@altskipped
\gre@nabc@voice@ii@altskippedfalse
% Shared flag: when true, \gre@nabc@prepare skips the left-overflow kern.
% Set and cleared by the deferred pre-phase around the \gre@nabc@prepare
% call so that each voice can independently suppress the kern.
\newif\ifgre@nabc@suppress@leftoverflow
\gre@nabc@suppress@leftoverflowfalse

\def\gresetnabc#1#2{%
  \gre@processvisibilitywithphantom{#2}%
    {\csname gre@nabcvoice@\romannumeral#1@visibletrue\endcsname}%
    {\csname gre@nabcvoice@\romannumeral#1@visiblefalse\endcsname}%
    {\csname gre@nabcvoice@\romannumeral#1@phantomwrapper\endcsname}%
    {\protect\gresetnabc}%
}

% See Command_Index_User.tex for documentation.
\def\gresetnabcalignment{\@ifnextchar[{\gre@setnabcalignment@voice}{\gre@setnabcalignment@global}}%
\def\gre@setnabcalignment@global#1{%
  \IfStrEqCase{#1}{%
    {neume}%
      {\directlua{gregoriotex.set_nabc_alignment("neume")}}%
    {full}%
      {\directlua{gregoriotex.set_nabc_alignment("full")}}%
    }[% all other cases
      \gre@error{Unrecognized option "#1" for \protect\gresetnabcalignment\MessageBreak Possible options are: 'neume' and 'full'}%
    ]%
}%
\def\gre@setnabcalignment@voice[#1]#2{%
  \IfStrEqCase{#2}{%
    {neume}%
      {\directlua{gregoriotex.set_nabc_alignment("neume", #1)}}%
    {full}%
      {\directlua{gregoriotex.set_nabc_alignment("full", #1)}}%
    }[% all other cases
      \gre@error{Unrecognized option "#2" for \protect\gresetnabcalignment\MessageBreak Possible options are: 'neume' and 'full'}%
    ]%
}

% See Command_Index_User.tex for documentation.
\def\gresetnabcskipalterations{%
  \@ifnextchar[{\gre@setnabcskipalterations@voice}{\gre@setnabcskipalterations@global}%
}%
\def\gre@setnabcskipalterations@global#1{%
  \IfStrEqCase{#1}{%
    {true}%
      {\global\gre@nabc@voice@i@skipalterationstrue
       \global\gre@nabc@voice@ii@skipalterationstrue}%
    {false}%
      {\global\gre@nabc@voice@i@skipalterationsfalse
       \global\gre@nabc@voice@ii@skipalterationsfalse}%
    }[%
      \gre@error{Unrecognized option "#1" for \protect\gresetnabcskipalterations\MessageBreak Possible options are: 'true' and 'false'}%
    ]%
}%
\def\gre@setnabcskipalterations@voice[#1]#2{%
  \IfStrEqCase{#2}{%
    {true}%
      {\ifnum#1=1\relax\global\gre@nabc@voice@i@skipalterationstrue\fi%
       \ifnum#1=2\relax\global\gre@nabc@voice@ii@skipalterationstrue\fi}%
    {false}%
      {\ifnum#1=1\relax\global\gre@nabc@voice@i@skipalterationsfalse\fi%
       \ifnum#1=2\relax\global\gre@nabc@voice@ii@skipalterationsfalse\fi}%
    }[%
      \gre@error{Unrecognized option "#2" for \protect\gresetnabcskipalterations\MessageBreak Possible options are: 'true' and 'false'}%
    ]%
}

\def\GreNABCNeumes#1#2#3#4{%
  \csname ifgre@nabcvoice@\romannumeral#1@visible\endcsname %
    \ifcase#1\relax
      \or % 1
        \csname GreSetNabcAboveLines%
          \ifgre@nabc@voice@i@skipalterations Deferred\fi\endcsname
        {\gre@applyphantomwrapper{\gre@nabcvoice@i@phantomwrapper}{\GreNABCChar[1]{#2}}}%
      \or % 2
        \csname GreSetNabcBelowLines%
          \ifgre@nabc@voice@ii@skipalterations Deferred\fi\endcsname
        {\gre@applyphantomwrapper{\gre@nabcvoice@ii@phantomwrapper}{\GreNABCChar[2]{#2}}}%
      \fi
  \fi %
}

% Flush pending NABC neumes for elements without glyphs (empty GABC snippet)

  \gre@nabc@pre@voice@ii
  \gre@nabc@emit@voice@i
  \gre@nabc@emit@voice@ii
}
