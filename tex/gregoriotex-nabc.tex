% \gresetnabcfont[#1]#2#3
% #1: voice
% #2: font name
% #3: font size
\def\gresetnabcfont{\@ifnextchar[{\gre@setnabcvoicefont}{\gre@setnabcvoicefont[1]}}%

\def\gre@setnabcvoicefont[#1]#2#3{%
  \expandafter\xdef\csname gre@nabcvoice@\romannumeral#1@fontname\endcsname{#2}%
  \expandafter\font\csname gre@font@nabcvoice@\romannumeral#1\endcsname={name:#2} at #3 pt\relax
  \global\font\gre@font@nabc={name:#2} at #3 pt\relax %
  {%
    \csname gre@font@nabcvoice@\romannumeral#1\endcsname
    \directlua{gregoriotex.init_nabc_font("\luatexluaescapestring{#2}")}%
  }%
}

\def\gre@nabccharno#1#2#3{%
  \gre@trace{gre@nabccharno{#1}{#2}{#3}}%
  {\directlua{gregoriotex.print_nabc(gregoriotex.parse_nabc("#1", "\luatexluaescapestring{#2}", #3))}}%
  \gre@trace@end%
}

% \GreNABCChar[#1]#2
% #1: voice
% #2: char
\def\GreNABCChar{\@ifnextchar[{\gre@nabcvoicechar}{\gre@nabcvoicechar[1]}}%
\def\gre@nabcvoicechar[#1]#2{{%
  %\catcode`\~=12{}%
  %\catcode`\@=11{}%
  \csname gre@font@nabcvoice@\romannumeral#1\endcsname
  \ifcase#1\relax\or\gre@style@nabc\or\gre@style@belowlinesnabc\fi
  \gre@nabccharno{#2}{\csname gre@nabcvoice@\romannumeral#1@fontname\endcsname}{1}%
  \ifcase#1\relax\or\endgre@style@nabc\or\endgre@style@belowlinesnabc\fi
}}

\newif\ifgre@nabcvoice@i@visible
\gre@nabcvoice@i@visibletrue
\newif\ifgre@nabcvoice@ii@visible
\gre@nabcvoice@ii@visibletrue
% define more of these when more voices are supported

\def\gresetnabc#1#2{%
  \IfStrEqCase{#2}{%
    {visible}%
      {\csname gre@nabcvoice@\romannumeral#1@visibletrue\endcsname}%
    {invisible}%
      {\csname gre@nabcvoice@\romannumeral#1@visiblefalse\endcsname}%
    }[% all other cases
      \gre@error{Unrecognized option "#2" for \protect\gresetnabc\MessageBreak Possible options are: 'visible' and 'invisible'}%
    ]%
}

\def\GreNABCNeumes#1#2#3#4{%
  \csname ifgre@nabcvoice@\romannumeral#1@visible\endcsname %
    \ifcase#1\relax
      \or % 1
        \GreSetNabcAboveLines{\GreNABCChar[1]{#2}}%
      \or % 2
        \GreSetNabcBelowLines{\GreNABCChar[2]{#2}}%
      \fi
  \fi %
}

\newif\ifgre@nabcfontloaded%
\gre@nabcfontloadedfalse%

\def\GreScoreNABCLines#1{%
  \ifcsname gre@font@nabcvoice@i\endcsname\else%
    \gresetnabcfont[1]{gregall}{8}%
  \fi
  \ifcsname gre@font@nabcvoice@ii\endcsname\else%
    \gresetnabcfont[2]{gregall}{8}%
  \fi
}
